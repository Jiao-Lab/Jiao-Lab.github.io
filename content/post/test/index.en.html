get_smr <- function(data,xname,yname,low,up){
  d <- as.data.frame(cbind(data[,xname],data[,yname]))
  colnames(d) <- c("x","y")
  
  smr <- d %>% 
    group_by(x) %>% 
    summarise(median(y),quantile(y,low),quantile(y,up))
  
  colnames(smr) <- c("time","q50","qlow","qup")
  
  smr
}
